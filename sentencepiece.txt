// src/sentencepiece.rs - Optimized SentencePiece tokenization with DAG-based Viterbi

use crate::{Result, TokenizerError, VOCAB, UNK_TOKEN_ID};
use ahash::{AHashMap, AHashSet};
use std::cmp::Ordering;
use std::f32;

const WHITESPACE_MARKER: char = '▁';
const BYTE_FALLBACK_PREFIX: &str = "<0x";
const MAX_PIECE_LENGTH: usize = 16; // T5 uses relatively short pieces

/// Trie node for efficient prefix matching
#[derive(Default)]
pub struct TrieNode {
    /// Token info: (token_id, score)
    pub token_info: Option<(u32, f32)>,
    /// Children nodes
    pub children: AHashMap<char, Box<TrieNode>>,
}

impl TrieNode {
    fn new() -> Self {
        Self {
            token_info: None,
            children: AHashMap::new(),
        }
    }
}

/// Optimized SentencePiece tokenizer using DAG-based Viterbi
pub struct SentencePieceTokenizer {
    /// Trie for efficient prefix matching
    trie: TrieNode,
    /// Vocabulary with scores
    vocab_scores: AHashMap<String, (u32, f32)>,
    /// Precomputed byte fallback tokens
    byte_fallback: [Option<u32>; 256],
}

impl SentencePieceTokenizer {
    /// Create a new SentencePiece tokenizer from vocabulary
    pub fn new(vocab: &phf::Map<&'static str, u32>) -> Self {
        let mut trie = TrieNode::new();
        let mut vocab_scores = AHashMap::with_capacity(vocab.len());
        let mut byte_fallback = [None; 256];
        
        // Build trie and score map
        for (token, &id) in vocab.entries() {
            // Compute score (using unigram probability approximation)
            // In production, these would come from the model file
            let score = Self::compute_token_score(token, id);
            vocab_scores.insert(token.to_string(), (id, score));
            
            // Insert into trie
            Self::insert_into_trie(&mut trie, token, id, score);
            
            // Check if it's a byte fallback token
            if let Some(byte_val) = Self::parse_byte_fallback(token) {
                byte_fallback[byte_val as usize] = Some(id);
            }
        }
        
        Self {
            trie,
            vocab_scores,
            byte_fallback,
        }
    }
    
    /// Tokenize text using optimized Viterbi algorithm
    pub fn tokenize(&self, text: &str) -> Result<Vec<u32>> {
        if text.is_empty() {
            return Ok(vec![]);
        }
        
        // Preprocess: add whitespace markers
        let processed = self.preprocess(text);
        let chars: Vec<char> = processed.chars().collect();
        let n = chars.len();
        
        // Dynamic programming arrays
        let mut best_score = vec![f32::NEG_INFINITY; n + 1];
        let mut best_path = vec![0; n + 1];
        let mut best_token_id = vec![UNK_TOKEN_ID; n + 1];
        
        // Initialize
        best_score[0] = 0.0;
        
        // Forward pass: build DAG
        for start_idx in 0..n {
            if best_score[start_idx] == f32::NEG_INFINITY {
                continue;
            }
            
            // Try all possible token matches starting at start_idx
            self.find_tokens_at_position(
                &chars,
                start_idx,
                &mut best_score,
                &mut best_path,
                &mut best_token_id,
            );
        }
        
        // Backward pass: reconstruct best path
        let mut tokens = Vec::new();
        let mut pos = n;
        
        while pos > 0 {
            tokens.push(best_token_id[pos]);
            pos = best_path[pos];
        }
        
        tokens.reverse();
        Ok(tokens)
    }
    
    /// Find all possible tokens starting at given position
    fn find_tokens_at_position(
        &self,
        chars: &[char],
        start: usize,
        best_score: &mut [f32],
        best_path: &mut [usize],
        best_token_id: &mut [u32],
    ) {
        let mut node = &self.trie;
        let max_end = (start + MAX_PIECE_LENGTH).min(chars.len());
        
        // Build token string incrementally for efficiency
        let mut token_str = String::with_capacity(MAX_PIECE_LENGTH);
        
        for end in start..max_end {
            let ch = chars[end];
            token_str.push(ch);
            
            // Check if this character continues a valid prefix
            if let Some(child) = node.children.get(&ch) {
                node = child;
                
                // Check if we have a complete token
                if let Some((token_id, score)) = node.token_info {
                    let candidate_score = best_score[start] + score;
                    
                    // Update if this path is better
                    if candidate_score > best_score[end + 1] {
                        best_score[end + 1] = candidate_score;
                        best_path[end + 1] = start;
                        best_token_id[end + 1] = token_id;
                    }
                }
            } else {
                // No valid continuation
                break;
            }
        }
        
        // Handle unknown character with byte fallback
        if start + 1 <= chars.len() {
            let ch = chars[start];
            let byte_val = (ch as u32 & 0xFF) as u8;
            
            if let Some(token_id) = self.byte_fallback[byte_val as usize] {
                // Use byte fallback token
                let candidate_score = best_score[start] - 10.0; // Penalty for byte fallback
                if candidate_score > best_score[start + 1] {
                    best_score[start + 1] = candidate_score;
                    best_path[start + 1] = start;
                    best_token_id[start + 1] = token_id;
                }
            } else {
                // Use UNK token as last resort
                let candidate_score = best_score[start] - 100.0; // Heavy penalty for UNK
                if candidate_score > best_score[start + 1] {
                    best_score[start + 1] = candidate_score;
                    best_path[start + 1] = start;
                    best_token_id[start + 1] = UNK_TOKEN_ID;
                }
            }
        }
    }
    
    /// Preprocess text by adding whitespace markers
    fn preprocess(&self, text: &str) -> String {
        let mut result = String::with_capacity(text.len() + 10);
        let mut prev_was_whitespace = true;
        
        for ch in text.chars() {
            if ch.is_whitespace() {
                prev_was_whitespace = true;
                result.push(' ');
            } else {
                if prev_was_whitespace && !result.is_empty() {
                    // Replace last space with marker + current char
                    result.pop();
                    result.push(WHITESPACE_MARKER);
                }
                result.push(ch);
                prev_was_whitespace = false;
            }
        }
        
        // Add initial whitespace marker if text starts with non-whitespace
        if !text.starts_with(char::is_whitespace) {
            let mut new_result = String::with_capacity(result.len() + 1);
            new_result.push(WHITESPACE_MARKER);
            new_result.push_str(&result);
            new_result
        } else {
            result
        }
    }
    
    /// Insert token into trie
    fn insert_into_trie(trie: &mut TrieNode, token: &str, id: u32, score: f32) {
        let mut node = trie;
        
        for ch in token.chars() {
            node = node.children
                .entry(ch)
                .or_insert_with(|| Box::new(TrieNode::new()));
        }
        
        node.token_info = Some((id, score));
    }
    
    /// Compute token score (approximation for demo)
    fn compute_token_score(token: &str, id: u32) -> f32 {
        // In real implementation, load from model
        // This is a simple heuristic: prefer longer tokens and lower IDs
        let length_bonus = (token.len() as f32).ln();
        let id_penalty = (id as f32).ln() * 0.01;
        length_bonus - id_penalty
    }
    
    /// Parse byte fallback token
    fn parse_byte_fallback(token: &str) -> Option<u8> {
        if token.starts_with(BYTE_FALLBACK_PREFIX) && token.ends_with('>') {
            let hex_str = &token[BYTE_FALLBACK_PREFIX.len()..token.len() - 1];
            u8::from_str_radix(hex_str, 16).ok()
        } else {
            None
        }
    }
}

// src/optimizations.rs - Memory layout optimizations

use std::alloc::{alloc, dealloc, Layout};
use std::mem;
use std::ptr;

/// Cache-aligned vocabulary entry for optimal memory access
#[repr(C, align(64))]
pub struct CacheAlignedVocabEntry {
    pub token: [u8; 32], // Fixed-size token storage
    pub token_len: u8,
    pub id: u32,
    pub score: f32,
    _padding: [u8; 23], // Pad to 64 bytes
}

/// Memory pool for vocabulary entries
pub struct VocabMemoryPool {
    ptr: *mut CacheAlignedVocabEntry,
    capacity: usize,
    len: usize,
    layout: Layout,
}

impl VocabMemoryPool {
    /// Create a new memory pool with given capacity
    pub fn new(capacity: usize) -> Self {
        let layout = Layout::array::<CacheAlignedVocabEntry>(capacity).unwrap();
        let ptr = unsafe { alloc(layout) as *mut CacheAlignedVocabEntry };
        
        if ptr.is_null() {
            panic!("Failed to allocate vocabulary memory pool");
        }
        
        Self {
            ptr,
            capacity,
            len: 0,
            layout,
        }
    }
    
    /// Add entry to pool
    pub fn add_entry(&mut self, token: &str, id: u32, score: f32) -> Option<&CacheAlignedVocabEntry> {
        if self.len >= self.capacity || token.len() > 32 {
            return None;
        }
        
        unsafe {
            let entry = self.ptr.add(self.len);
            
            // Initialize entry
            ptr::write_bytes((*entry).token.as_mut_ptr(), 0, 32);
            ptr::copy_nonoverlapping(
                token.as_bytes().as_ptr(),
                (*entry).token.as_mut_ptr(),
                token.len(),
            );
            
            (*entry).token_len = token.len() as u8;
            (*entry).id = id;
            (*entry).score = score;
            
            self.len += 1;
            Some(&*entry)
        }
    }
    
    /// Get entry by index
    pub fn get(&self, index: usize) -> Option<&CacheAlignedVocabEntry> {
        if index < self.len {
            unsafe { Some(&*self.ptr.add(index)) }
        } else {
            None
        }
    }
}

impl Drop for VocabMemoryPool {
    fn drop(&mut self) {
        unsafe {
            dealloc(self.ptr as *mut u8, self.layout);
        }
    }
}

// Safety: Pool manages its own memory and provides safe access
unsafe impl Send for VocabMemoryPool {}
unsafe impl Sync for VocabMemoryPool {}

// src/example.rs - Complete usage example

use anyhow::Result;
use candle_core::{Device, Tensor};
use flan_t5_tokenizer::{
    FlanT5Tokenizer, BatchTokenizer, BatchConfig, 
    TokenizerConfig, TokenizerCandle
};

fn main() -> Result<()> {
    // 1. Basic tokenization
    basic_tokenization_example()?;
    
    // 2. Batch processing
    batch_processing_example()?;
    
    // 3. Candle integration
    candle_integration_example()?;
    
    // 4. Performance optimization
    performance_tuning_example()?;
    
    Ok(())
}

fn basic_tokenization_example() -> Result<()> {
    println!("=== Basic Tokenization Example ===");
    
    let tokenizer = FlanT5Tokenizer::with_default_config();
    
    let texts = vec![
        "Hello, world!",
        "The quick brown fox jumps over the lazy dog.",
        "Machine learning is fascinating.",
        "🦀 Rust is awesome! 🚀",
    ];
    
    for text in texts {
        let tokens = tokenizer.encode(text)?;
        let decoded = tokenizer.decode(&tokens)?;
        
        println!("Original: {}", text);
        println!("Tokens: {:?}", tokens);
        println!("Decoded: {}", decoded);
        println!();
    }
    
    Ok(())
}

fn batch_processing_example() -> Result<()> {
    println!("=== Batch Processing Example ===");
    
    let config = BatchConfig {
        max_batch_size: 64,
        batch_timeout: std::time::Duration::from_millis(5),
        num_workers: 4,
    };
    
    let tokenizer = FlanT5Tokenizer::with_default_config();
    let batch_tokenizer = BatchTokenizer::new(tokenizer, config);
    
    // Generate batch of texts
    let texts: Vec<String> = (0..100)
        .map(|i| format!("This is sentence number {}. It contains some text for tokenization.", i))
        .collect();
    let text_refs: Vec<&str> = texts.iter().map(|s| s.as_str()).collect();
    
    let start = std::time::Instant::now();
    let results = batch_tokenizer.encode_batch(&text_refs)?;
    let duration = start.elapsed();
    
    println!("Tokenized {} texts in {:?}", texts.len(), duration);
    println!("Average time per text: {:?}", duration / texts.len() as u32);
    println!("First result: {:?}", &results[0]);
    
    Ok(())
}

fn candle_integration_example() -> Result<()> {
    println!("=== Candle Integration Example ===");
    
    let device = Device::Cpu;
    let mut tokenizer_config = TokenizerConfig::default();
    tokenizer_config.pad_to_max_length = true;
    tokenizer_config.max_length = 128;
    
    let tokenizer = FlanT5Tokenizer::new(tokenizer_config);
    
    // Single text to tensor
    let text = "Translate English to German: Hello, how are you?";
    let tensors = tokenizer.tokenize_to_tensor(text, &device)?;
    
    println!("Single text tensor shapes:");
    println!("  Input IDs: {:?}", tensors.input_ids.shape());
    println!("  Attention mask: {:?}", tensors.attention_mask.shape());
    
    // Batch texts to tensor
    let batch_texts = vec![
        "Translate English to French: Good morning",
        "Translate English to Spanish: Thank you",
        "Translate English to Italian: Goodbye",
    ];
    
    let batch_tensors = tokenizer.batch_tokenize_to_tensor(&batch_texts, &device)?;
    
    println!("\nBatch tensor shapes:");
    println!("  Input IDs: {:?}", batch_tensors.input_ids.shape());
    println!("  Attention mask: {:?}", batch_tensors.attention_mask.shape());
    
    Ok(())
}

fn performance_tuning_example() -> Result<()> {
    println!("=== Performance Tuning Example ===");
    
    // Create tokenizer with cache
    let tokenizer = FlanT5Tokenizer::with_default_config();
    
    // Warm up cache
    let warmup_texts = vec![
        "The quick brown fox",
        "Machine learning model",
        "Natural language processing",
    ];
    
    for text in &warmup_texts {
        tokenizer.encode(text)?;
    }
    
    // Benchmark with cache hits
    let iterations = 10_000;
    let start = std::time::Instant::now();
    
    for i in 0..iterations {
        let text = &warmup_texts[i % warmup_texts.len()];
        tokenizer.encode(text)?;
    }
    
    let duration = start.elapsed();
    println!("Cached tokenization: {} iterations in {:?}", iterations, duration);
    println!("Average time: {:?}", duration / iterations);
    
    // Clear cache and benchmark without cache
    tokenizer.clear_cache();
    
    let start = std::time::Instant::now();
    for i in 0..iterations {
        let text = format!("Unique text number {}", i);
        tokenizer.encode(&text)?;
    }
    
    let duration = start.elapsed();
    println!("\nUncached tokenization: {} iterations in {:?}", iterations, duration);
    println!("Average time: {:?}", duration / iterations);
    
    Ok(())
}

#[cfg(test)]
mod sentencepiece_tests {
    use super::*;
    
    #[test]
    fn test_whitespace_preprocessing() {
        let tokenizer = create_test_tokenizer();
        
        // Test cases
        let cases = vec![
            ("hello world", "▁hello▁world"),
            ("  spaces  ", "▁spaces"),
            ("one", "▁one"),
            ("", ""),
        ];
        
        for (input, expected_prefix) in cases {
            let processed = tokenizer.preprocess(input);
            assert!(
                processed.starts_with(expected_prefix) || processed == expected_prefix,
                "Input: '{}', Expected prefix: '{}', Got: '{}'",
                input, expected_prefix, processed
            );
        }
    }
    
    #[test]
    fn test_viterbi_tokenization() {
        let tokenizer = create_test_tokenizer();
        
        let text = "Hello world";
        let tokens = tokenizer.tokenize(text).unwrap();
        
        // Should produce non-empty token sequence
        assert!(!tokens.is_empty());
        
        // All tokens should be valid (not UNK unless necessary)
        for &token_id in &tokens {
            assert!(token_id < VOCAB_SIZE as u32);
        }
    }
    
    fn create_test_tokenizer() -> SentencePieceTokenizer {
        SentencePieceTokenizer::new(&VOCAB)
    }
